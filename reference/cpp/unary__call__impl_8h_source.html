110  enum class ReleaseTrySend
111  {
112  No,
113  Yes
114  };

116  void AddRef() noexcept
117  {
118  _refCount.fetch_add(1, std::memory_order_relaxed);
119  }
120 
121  void Release(ReleaseTrySend trySend)
122  {
123  auto oldCount = _refCount.fetch_sub(1, std::memory_order_acq_rel);
124 
125  switch (oldCount)
126  {
127  case 2:
128  // There are two possible cases when the old count can be 2.
129  //
130  // Case 2a (trySend == ReleaseTrySend::Yes): the last
131  // user reference has just gone away, but Finish was not
132  // called. In this case, we are responsible for sending
133  // an error response and decrementing the final ref
134  // count. FinishWithError will schedule the send of the
135  // error response, and notification of completion of the
136  // send via invoke() will decrement the final ref count.
137  // Since we hold the ref count ourselves, we will not
138  // get deleted until we're done sending.
139  //
140  // Case 2b (trySend == ReleaseTrySend::No): the user
141  // called Finish/FinishWithError and completion of
142  // sending that response has just happened via invoke(),
143  // AND there is still ONE user reference. In this case,
144  // the user reference could go away at any time and
145  // delete us, so we can't touch any part of this object.
146  // But we don't need to send anything anyway.
147  if (trySend == ReleaseTrySend::Yes)
148  {
149  FinishWithError({ grpc::StatusCode::INTERNAL, "An internal server error has occurred." });
150  }
151  break;
152 
153  case 1:
154  delete this;
155  break;
156 
157  default:
158  break;
159  }
160  }
