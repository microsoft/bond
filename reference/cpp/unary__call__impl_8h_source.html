85 template <typename T = Void>
86 void Finish(const T& response = {})
87 {

91 template <typename T>
92 void Finish(const bonded<T>& response)
93 {
94 bool wasResponseSent = _responseSentFlag.test_and_set();
95 if (!wasResponseSent)
96 {
97 _responder.Finish(Serialize(response), ::grpc::Status::OK, tag());
98 }
99 }
100
101 void Finish(const ::grpc::Status& status)
102 {
103 bool wasResponseSent = _responseSentFlag.test_and_set();
104 if (!wasResponseSent)
105 {
106 _responder.FinishWithError(status, tag());
107 }
108 }
109
110 private:
111 void invoke(bool /* ok */) override
112 {
113 // The response has been sent, so we no longer need to keep
114 // ourselves alive: release the implicit initial refcount that
115 // this instance was constructed with.
116 Release();
117 }
118
119 void AddRef() noexcept
120 {
121 _refCount.fetch_add(1, std::memory_order_relaxed);
122 }
123
124 void Release()
125 {
126 if (_refCount.fetch_sub(1, std::memory_order_acq_rel) == 1)
127 {
128 delete this;
129 }
130 }
131
132 void TryFinishWithError()
133 {
134 if (_refCount.load(std::memory_order::memory_order_acquire) == 2)
135 {
136 // The last user reference has just gone away, but Finish was
137 // not called. In this case, we are responsible for sending
138 // an error response and decrementing the final ref
139 // count. Finish with ::grpc::Status will schedule the send of the
140 // error response, and notification of completion of the
141 // send via invoke() will decrement the final ref count.
142 // Since we hold the ref count ourselves, we will not
143 // get deleted until we're done sending.
144 //
145 // Even when multiple threads enter this case, at most one
146 // will succeed in sending error, thanks to _responseSentFlag.
147
148 Finish({ ::grpc::StatusCode::INTERNAL, "An internal server error has occurred." });
149 }
150 }
