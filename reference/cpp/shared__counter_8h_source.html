189  {
190  single_threaded_counter<value_type> counter{ max_value };
191  capped_allocator<Alloc, decltype(counter)&> capped_alloc{ std::ref(counter), alloc };
192  shared_counter shared_counter{
193  max_value,
194  capped_alloc }; // Only a copy of underlying \c Alloc will be stored as part of
195  // the counter, so it is safe to have the \c counter on the stack.
196  shared_counter.try_add(counter.value());
