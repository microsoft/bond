282 inline blob merge(const A& allocator, const blob& x, const blob& y)
283 {
284  if (x.empty() || y.empty())
285  {

304 template <typename t_It, typename A>
305 inline blob merge(const A& allocator, t_It begin, t_It end)
306 {
307  //
308  // calculate the size of resulting blob
309  //
310  uint32_t length = 0;
311  for (t_It it = begin; it != end; ++it)
312  {
313  length += it->length();
314  }
315 
316  if (0 == length)
317  {
318  //
319  // empty blob to return
320  //
321  return blob();
322  }
323  else
324  if (length == begin->length())
325  {
326  //
327  // just first blob in the sequence is not empty
328  //
329  return *begin;
330  }
331  else
332  {
333  //
334  // sequence of several non-empty blobs
335  //
336  BOOST_ASSERT(length > begin->length());
337

339
