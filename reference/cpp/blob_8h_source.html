312 inline blob merge(const A& allocator, const blob& x, const blob& y)
313 {
314  if (x.empty() || y.empty())
315  {
316  //
317  // one of provided blobs is empty,
318  // return the other one
319  //

321  }

334 template <typename t_It, typename A>
335 inline blob merge(const A& allocator, t_It begin, t_It end)
336 {
337  //
338  // calculate the size of resulting blob
339  //
340  uint32_t length = 0;
341  for (t_It it = begin; it != end; ++it)
342  {
343  length = detail::checked_add(length, it->length());
344  }
345 
346  if (0 == length)
347  {
348  //
349  // empty blob to return
350  //
351  return blob();
352  }
353  else if (length == begin->length())
354  {
355  //
356  // just first blob in the sequence is not empty
357  //
358  return *begin;
359  }
360  else
361  {
362  //
363  // sequence of several non-empty blobs
364  //
365  BOOST_ASSERT(length > begin->length());
366 
367  boost::shared_ptr<char[]> buffer = boost::allocate_shared_noinit<char[]>(allocator, length);
368 
369  uint32_t offset = 0;
370  for (t_It it = begin; it != end; ++it)
371  {
372  ::memcpy(buffer.get() + offset, it->content(), it->length());
373 
374  offset += it->length();
375  }
376 
377  BOOST_ASSERT(offset == length);
378  return blob(buffer, length);
379  }
