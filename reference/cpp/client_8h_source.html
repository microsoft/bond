111 {
112 auto callback = std::make_shared<std::packaged_task<unary_call_result<Response>(unary_call_result<Response>)>>(
113 [](unary_call_result<Response> response)
114 {
115 return response.status().ok()
116 ? response

123 [callback](unary_call_result<Response> response)

125 (*callback)(std::move(response));
126 };

128 dispatch(method, std::move(context), copyableCallback, request);

130 return result;
