368 inline
369 const T& mapped_at(const std::map<K, T, C, A>& map, const K& key)
370 {
371  typename std::map<K, T, C, A>::const_iterator it = map.find(key);
372 
373  if (it == map.end())

376  return it->second;

453 template <typename K, typename V>
454 std::map<V, K> reverse_map(const std::map<K, V>& map)
455 {
456  std::map<V, K> reversed;
457 
458  for (typename std::map<K, V>::const_iterator it = map.begin(); it != map.end(); ++it)
459  reversed[it->second] = it->first;
