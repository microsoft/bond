13 class RandomProtocolEngine
14 {
15 public:
16  // We don't need a good pseudo-random number generator but we do need one
17  // that is consistent accross compilers/libraries so that we can use
18  // randomly generated data files from one platfom to verify compatibility
19  // on another platform.
20  RandomProtocolEngine()
21  {}
22 
23  // Variant of Marsaglia’s xorshift generator
24  // http://arxiv.org/pdf/1404.0390v1.pdf
25  uint64_t Next()
26  {
27  uint64_t s1 = state[0];
28  const uint64_t s0 = state[1];
29  state[0] = s0;
30  s1 ^= s1 << 23;
31  return (state[1] = (s1 ^ s0 ^ (s1 >> 17) ^ (s0 >> 26))) + s0;
32  }
33 
34  void Seed(uint64_t s0 = seed1, uint64_t s1 = seed2)
35  {
36  state[0] = s0;
37  state[1] = s1;
38  }
39 
40  static uint64_t state[2];
41  const static uint64_t seed1 = 234578354U;
42  const static uint64_t seed2 = 753478U;
43 };

105  void Read(double& value)
106  {
107  uint8_t sign;
108  uint8_t exponent;
109  uint32_t mantissa;
110 
111  Read(sign);
112  Read(exponent);
113  Read(mantissa);
114 
115  // don't return special values: infinity, NaN
116  if (exponent == 0)
117  exponent = 0x80;
118 
119  uint64_t bits = ((uint64_t)(sign) << 63) | ((uint64_t)(exponent) << (52 + 3)) | (uint64_t)mantissa;
120 
121  *reinterpret_cast<uint64_t*>(&value) = bits;
122  }

124  void Read(float& value)

126  uint8_t sign;
127  uint8_t exponent;
128  uint16_t mantissa;
129 
130  Read(sign);
131  Read(exponent);
132  Read(mantissa);
133 
134  // don't return special values: infinity, NaN
135  if (exponent == 0 || exponent == 0xff)
136  exponent = 0x80;
137 
138  uint32_t bits = ((uint32_t)(sign) << 31) | ((uint32_t)(exponent) << 23) | (uint32_t)mantissa;
