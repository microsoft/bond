26 // Adapter from Bond input stream to rapidjson read-only stream
27 template <typename Buffer>
28 class RapidJsonInputStream
29 {
30 public:
31  typedef char Ch;
32 
33  RapidJsonInputStream(typename boost::call_traits<Buffer>::reference input)
34  : input(&input),
35  current(0),
36  count(0)
37  {
38  input.Read(current);
39  }
40 
41  RapidJsonInputStream(const RapidJsonInputStream& that, typename boost::call_traits<Buffer>::reference input)
42  : input(&input),
43  current(that.current),
44  count(that.count)
45  {}
46 
47  char Peek()
48  {
49  if (!current)
50  input->Read(current);
51 
52  return current;
53  }
54 
55  size_t Tell() const
56  {
57  return count;
58  }
59 
60  char Take()
61  {
62  char c = current;
63  current = '\0';
64 
65  if (!c)
66  input->Read(c);
67 
68  ++count;
69  return c;
70  }
71 
72  // not implemented for read only stream
73  char* PutBegin() { BOOST_ASSERT(false); return 0; }
74  void Put(char) { BOOST_ASSERT(false); }
75  size_t PutEnd(char*) { BOOST_ASSERT(false); return 0; }
76 
77  RapidJsonInputStream& operator=(const RapidJsonInputStream& that)
78  {
79  // rapidjson reader makes a local copy of stream within some functions
80  // and assigns it back to its main stream variable before function exit.
81  BOOST_ASSERT(input == that.input);
82  current = that.current;
83  count = that.count;
84  return *this;
85  }
86 
87 private:
88  Buffer* input;
89  uint8_t current;
90  size_t count;
91 };

135 class JsonTypeMatching : boost::noncopyable
136 {
137 public:
138  JsonTypeMatching(BondDataType type, BondDataType schema, bool is_enum)
139  : matchesObject(type == BT_STRUCT && type == schema),
140  matchesArray((type == BT_MAP || type == BT_LIST || type == BT_SET) && type == schema),
141  matchesNull(type == BT_LIST && type == schema),
142  matchesInt(type >= BT_INT8 && type <= BT_INT64),
143  matchesInt64(type == BT_INT64),
144  matchesUint(type >= BT_UINT8 && type <= BT_UINT64),
145  matchesUint64(type == BT_UINT64),
146  matchesNumber(type >= BT_FLOAT && type <= BT_DOUBLE),
147  matchesString(type == BT_STRING || type == BT_WSTRING || is_enum),
148  matchesBool(type == BT_BOOL)
149  {}

151 
152  bool TypeMatch(const rapidjson::Value& value) const
153  {
154  return ComplexTypeMatch(value) || BasicTypeMatch(value);
155  }
156 
157  bool ComplexTypeMatch(const rapidjson::Value& value) const
158  {
159  return ((value.IsObject() && matchesObject)
160  || (value.IsArray() && matchesArray)
161  || (value.IsNull() && matchesNull));
162  }
163 
164  bool BasicTypeMatch(const rapidjson::Value& value) const
165  {
166  return ((value.IsString() && matchesString)
167  || (value.IsUint() && matchesUint)
168  || (value.IsInt() && matchesInt)
169  || (value.IsUint64() && matchesUint64)
170  || (value.IsInt64() && matchesInt64)
171  || (value.IsNumber() && matchesNumber)
172  || (value.IsBool() && matchesBool));
173  }
174 
175 private:
176  const bool matchesObject;
177  const bool matchesArray;
178  const bool matchesNull;
179  const bool matchesInt;
180  const bool matchesInt64;
181  const bool matchesUint;
182  const bool matchesUint64;
183  const bool matchesNumber;
184  const bool matchesString;
185  const bool matchesBool;
186 };

233 Read(const rapidjson::Value& value, T& var)

242 // wstring
243 template <typename T>
244 typename boost::enable_if<is_wstring<T> >::type
245 Read(const rapidjson::Value& value, T& var)
246 {
247  std::basic_string<uint16_t> str =
248  boost::locale::conv::utf_to_utf<uint16_t>(
249  value.GetString(), value.GetString() + value.GetStringLength(), boost::locale::conv::stop);
250  const uint32_t length = static_cast<uint32_t>(str.size());
251 
252  resize_string(var, length);
253  std::copy(str.begin(), str.end(), string_data(var));
254 }
