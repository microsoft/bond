67 template <typename Callback>
68 void operator()(Callback&& callback)
69 {

74 enum class pool_type
75 {
76 new_instance,
77 default_instance
78 };
79
80 class impl
81 {
82 public:
83 impl(pool_type type, boost::optional<std::pair<uint32_t, uint32_t>> numThreads)
84 : _env{ nullptr, ::DestroyThreadpoolEnvironment },
85 _pool{ nullptr, ::CloseThreadpool },
86 _group{ nullptr, ::CloseThreadpoolCleanupGroup }
87 {
88 ::InitializeThreadpoolEnvironment(&_envInst);
89 _env.reset(&_envInst);
90
91 if (type == pool_type::new_instance)
92 {
93 _pool.reset(::CreateThreadpool(nullptr));
94 if (!_pool)
95 {
96 detail::Win32Exception("Failed to create thread pool.");
97 }
98
99 ::SetThreadpoolCallbackPool(_env.get(), _pool.get());
100 }
101
102 _group.reset(::CreateThreadpoolCleanupGroup());
103 if (!_group)
104 {
105 detail::Win32Exception("Failed to create thread pool cleanup group.");
106 }
107
108 ::SetThreadpoolCallbackCleanupGroup(_env.get(), _group.get(), nullptr);
109
110 if (numThreads)
111 {
112 SetThreadCount(static_cast<DWORD>(numThreads->first), static_cast<DWORD>(numThreads->second));
113 }
114 }
115
116 ~impl()
117 {
118 // Wait for all callbacks to return without canceling pending ones.
119 ::CloseThreadpoolCleanupGroupMembers(_group.get(), FALSE, nullptr);
120 }
121
122 impl(const impl& other) = delete;
123 impl& operator=(const impl& other) = delete;

125 template <typename Callback>
126 void schedule(Callback&& callback)
127 {
128 using Func = typename std::decay<Callback>::type;
129
130 std::unique_ptr<Func> func{ new Func{ std::forward<Callback>(callback) } };
131
132 if (!::TrySubmitThreadpoolCallback(
133 static_cast<::PTP_SIMPLE_CALLBACK>([](::PTP_CALLBACK_INSTANCE, PVOID context)
134 {
135 (*std::unique_ptr<Func>{ static_cast<Func*>(context) })();
136 }),
137 func.get(),
138 _env.get()))
139 {
140 detail::Win32Exception("Failed to submit thread pool callback.");
141 }
142
143 func.release();
144 }
145
146 private:
147 void SetThreadCount(DWORD minThreads, DWORD maxThreads)
148 {
149 if (!::SetThreadpoolThreadMinimum(_pool.get(), static_cast<DWORD>(minThreads)))
150 {
151 detail::Win32Exception("Failed to set thread pool minimum number of threads.");
152 }
153
154 ::SetThreadpoolThreadMaximum(_pool.get(), static_cast<DWORD>(maxThreads));
155 }
156
157 ::TP_CALLBACK_ENVIRON _envInst;
158 std::unique_ptr<::TP_CALLBACK_ENVIRON, decltype(&::DestroyThreadpoolEnvironment)> _env;
159 std::unique_ptr<::TP_POOL, decltype(&::CloseThreadpool)> _pool;
160 std::unique_ptr<::TP_CLEANUP_GROUP, decltype(&::CloseThreadpoolCleanupGroup)> _group;
161 };

167 {}

169 std::shared_ptr<impl> _impl;
170 };
