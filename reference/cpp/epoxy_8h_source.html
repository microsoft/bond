152  {

154  async_write(m_socket,

170  //
171  // This method is called to send next packet in the queue.
172  //
173  void DoSend()
174  {
175  do
176  {
177  BOOST_ASSERT(m_sending.load());
178 
179  uint64_t id = 0;
180  std::shared_ptr<const std::vector<blob>> buffers;
181 
182  {
183  auto l = lock(m_queue);
184  if (!l->empty())
185  {
186  auto& front = l->front();
187 
188  id = front.first;
189  buffers.swap(front.second);
190  l->pop_front();
191  }
192  }
193 
194  if (buffers != nullptr)
195  {
196  Write(id, buffers);
197  break;
198  }
199 
200  BOOST_VERIFY(m_sending.exchange(false));
201  }
202  while (!lock(m_queue)->empty() && !m_sending.exchange(true));
203  }

206  bool HandleFrame(bool success, detail::ClassifyResult&& result)
207  {
208  uint64_t conversation_id = UINT64_MAX;
209  if (nullptr != result.headers)
210  {
211  conversation_id = result.headers->conversation_id;
212  }
213 
214  Packet packet;
215  if (success)
216  {
217  if (detail::FrameDisposition::ProcessConfig == result.disposition)
218  {
219  BOND_LOG(LOG_ERROR, "HandleFrame",
220  "HandleFrame with handshake");
221 
222  if (m_handshake_received.exchange(true))
223  {
224  BOND_LOG(LOG_ERROR, "HandleFrame",
225  "HandleFrame called with duplicate handshake!");
226 
227  return false;
228  }
229 
230  if (!m_isClient)
231  {
232  DoHandshake();
233  }
234 
235  return true;
236  }
237 
238  if (result.disposition == detail::FrameDisposition::Indeterminate)
239  {
240  BOND_LOG(LOG_ERROR, "HandleFrame",
241  "HandleFrame called with success=true but with no result!");
242 
243  return false;
244  }
245 
246  if (nullptr == result.headers)
247  {
248  BOND_LOG(LOG_ERROR, "HandleFrame",
249  "HandleFrame called with null headers!");
250 
251  return false;
252  }
253 
254  if (!m_handshake_received.load())
255  {
256  BOND_LOG(LOG_ERROR, "HandleFrame",
257  "Handshake has not arrived!");
258 
259  return false;
260  }
261 
262  packet.method_name = std::move(result.headers->method_name);
263  packet.service_name = std::move(result.headers->service_name);
264  packet.layers = std::move(result.layer_data);
265  packet.message_data = std::move(result.message_data);
266 
267  //
268  // Execute appropriate handler against de-serialized packet.
269  //
270  switch (result.headers->message_type)
271  {
272  case EpoxyMessageType::REQUEST:
273  {
274  packet.message_type = MessageType::REQUEST;
275  break;
276  }
277 
278  case EpoxyMessageType::EVENT:
279  {
280  packet.message_type = MessageType::EVENT;
281  break;
282  }
283 
284  case EpoxyMessageType::RESPONSE:
285  {
286  packet.message_type = MessageType::RESPONSE;
287  break;
288  }
289 
290  default:
291  {
292  BOND_LOG(LOG_ERROR, "HandleFrame",
293  "HandleFrame called with unexpected message type: " << result.headers->message_type);
294 
295  return false;
296  }
297  }
298  }
299 
300  m_handler->PacketReceived(success, conversation_id, std::move(packet));
301  return true;
302  }

304  //
305  // This method is called by io_service once a framelet has been read.
306  //
307  void OnReadFramelet(const std::shared_ptr<detail::FrameState>& state,
308  const boost::system::error_code& ec)
309  {
310  BOOST_ASSERT(m_reading.load());
311 
312  if (ec)
313  {
314  //
315  // TODO: clarify possible errors and proper handling:
316  // && boost::asio::error::connection_reset == ec
317  // && boost::asio::error::not_connected != ec
318  // && boost::asio::error::misc_errors::eof != ec)
319  //
320  BOND_LOG(LOG_ERROR, "EpoxyConnection",
321  "Unexpected error from " << m_address << ": " << ec.message());
322 
323  detail::ClassifyResult empty;
324  HandleFrame(false, std::move(empty));
325 
326  DoDisconnect();
327  }
328  else
329  {
330  BOND_LOG(LOG_DEBUG, "EpoxyConnection",
331  "Data from " << m_address << " received");
332 
333  if (state->count > state->framelets.size())
334  {
335  DoReadFrameletTag(state);
336  }
337  else
338  {
339  std::unique_ptr<ClassifyResult> result = Classify(*state);
340  if (HandleFrame(true, std::move(*result)))
341  {
342  DoReadFrame(state);
343  }
344  else
345  {
346  DoDisconnect();
347  }
348  }
349  }
350  }

361  state->framelets.emplace_back(static_cast<detail::FrameletType>(state->tag.type), blob(data, length));
362 
363  boost::weak_ptr<EpoxyConnection> ptr = this->shared_from_this();
364  async_read(

367  [ptr, state](const boost::system::error_code& ec, std::size_t s) {
368  BOOST_VERIFY(!!ec || s == state->tag.length);
369 
370  if (auto this_ = ptr.lock())
371  {
372  this_->OnReadFramelet(state, ec);
373  }
374  });
375  }

474  //
475  // This method is called by io_service once connection was established.
476  //
477  void OnConnect(const boost::system::error_code& ec)
478  {
479  BOOST_ASSERT(m_isClient);
480  BOOST_ASSERT(m_sending.load());
481  BOOST_ASSERT(m_reading.load());
482 
483  if (!ec)
484  {
485  //
486  // Successfully connected
487  //
488  BOND_LOG(LOG_INFO, "EpoxyConnection",
489  "Connected to " << m_address);
490 
491  {
492  //
493  // Config tcp/ip socket options
494  //
495  boost::system::error_code so_ec;
496  if (m_socket.set_option(boost::asio::socket_base::keep_alive(true), so_ec))
497  {
498  BOND_LOG(LOG_ERROR, "EpoxyConnection",
499  "Failed to set keep_alive for a socket to " << m_address << ": " << so_ec.message());
500  }
501 
502  if (m_socket.set_option(boost::asio::ip::tcp::no_delay(true), so_ec))
503  {
504  BOND_LOG(LOG_ERROR, "EpoxyConnection",
505  "Failed to set no_delay for a socket to " << m_address << ": " << so_ec.message());
506  }
507 
508  if (m_socket.set_option(boost::asio::socket_base::linger(false, 0), so_ec))
509  {
510  BOND_LOG(LOG_ERROR, "EpoxyConnection",
511  "Failed to set linger for a socket to " << m_address << ": " << so_ec.message());
512  }
513  }
514 
515  //
516  // Reset local handshake and start remote handshake
517  //
518  m_handshake_received.exchange(false);
519  DoHandshake();
520 
521  //
522  // Schedule async read.
523  //
524  auto state = std::allocate_shared<detail::FrameState>(m_allocator);
525  DoReadFrame(state);
526  }
527  else
528  {
529  BOND_LOG(LOG_ERROR, "EpoxyConnection",
530  "Failed to connect to " << m_address << ": " << ec.message());
531 
532  BOOST_VERIFY(m_sending.exchange(false));
533 
534  DoDisconnect();
535  }
536  }

590  std::shared_ptr<std::vector<blob>> Frame(uint64_t id, const Packet& packet)
591  {
592  const bool have_layer_data = !packet.layers.empty();
593 
594  OutputBuffer stream(256, static_cast<uint32_t>(packet.layers.size() + packet.message_data.data.size() + 4));
595  stream.Write(static_cast<uint16_t>(have_layer_data ? 3 : 2));
596 
597  // Headers
598  {
599  epoxy::EpoxyHeaders headers;
600 
601  switch (packet.message_type)
602  {
603  case MessageType::EVENT: headers.message_type = epoxy::EpoxyMessageType::EVENT; break;
604  case MessageType::REQUEST: headers.message_type = epoxy::EpoxyMessageType::REQUEST; break;
605  case MessageType::RESPONSE: headers.message_type = epoxy::EpoxyMessageType::RESPONSE; break;
606 
607  default: BOND_THROW(TransportException, "Unexpected message_type value " << packet.message_type);
608  }
609 
610  headers.conversation_id = id;
611  headers.service_name = packet.service_name;
612  headers.method_name = packet.method_name;
613 
614  OutputBuffer output(256, 4);
615  FastBinaryWriter<OutputBuffer> writer(output);
616 
617  Serialize(headers, writer);
618 
619  std::vector<blob> data;
620  output.GetBuffers(data);
621 
622  uint32_t length = 0;
623  for (const blob& buffer : data)
624  {
625  length += buffer.length();
626  }
627 
628  stream.Write(static_cast<uint16_t>(detail::FrameletType::EPOXY_HEADERS));
629  stream.Write(static_cast<uint32_t>(length));
630 
631  for (const blob& buffer : data)
632  {
633  stream.Write(buffer);
634  }
635  }
636 
637  // Layers
638  if (have_layer_data)
639  {
640  uint32_t length = 0;
641  for (const blob& buffer : packet.layers)
642  {
643  length += buffer.length();
644  }
645 
646  stream.Write(static_cast<uint16_t>(detail::FrameletType::LAYER_DATA));
647  stream.Write(static_cast<uint32_t>(length));
648 
649  for (const blob& buffer : packet.layers)
650  {
651  stream.Write(buffer);
652  }
653  }
654 
655  // Message data
656  {
657  uint32_t length = 0;
658  for (const blob& buffer : packet.message_data.data)
659  {
660  length += buffer.length();
661  }
662 
663  FrameletType frameletType = (packet.message_data.is_error
664  ? detail::FrameletType::ERROR_DATA
665  : detail::FrameletType::PAYLOAD_DATA);
666  stream.Write(static_cast<uint16_t>(frameletType));
667  stream.Write(static_cast<uint32_t>(length));
668 
669  for (const blob& buffer : packet.message_data.data)
670  {
671  stream.Write(buffer);
672  }
673  }
674 
675  std::shared_ptr<std::vector<blob>> buffers =
676  std::allocate_shared<std::vector<blob>>(m_allocator);
677 
678  stream.GetBuffers(*buffers);
679  return buffers;
680  }

856  void Send(uint64_t id, Packet packet) override
857  {
858  BOND_LOG(LOG_DEBUG, "EpoxyConnection",
859  "Send packet (id: " << id << ") to " << m_address);
860 
861  std::shared_ptr<const std::vector<blob>> buffers = Frame(id, packet);
862 
863  if (m_sending.exchange(true))
864  {
865  BOND_LOG(LOG_DEBUG, "EpoxyConnection",
866  "There is an outstanding send, enqueue packet (id: " << id << ") to be send to " << m_address);
867 
868  //
869  // There is existing send in-progress, enqueue this frame
870  //
871  auto l = lock(m_queue);
872  l->emplace_back(0, std::move(buffers));
873 
874  //
875  // Try to start send if previous send finished by this time
876  //
877  if (!m_sending.exchange(true))
878  {
879  DoSend();
880  }
881  }
882  else if (m_reading.exchange(true))
883  {
884  BOND_LOG(LOG_DEBUG, "EpoxyConnection",
885  "No outstanding sends, write packet (id: " << id << ") directly to " << m_address);
886 
887  //
888  // There is no existing send in-progress, try to send within this thread
889  //
890  Write(id, buffers);
891  }
892  else if (m_reconnect)
893  {
894  BOND_LOG(LOG_DEBUG, "EpoxyConnection",
895  "Connection for packet (id: " << id << ") is not established to " << m_address << ", start reconnect");
896 
897  //
898  // There is no existing connection, enqueue this frame and start connect
899  //
900  auto l = lock(m_queue);
901  l->emplace_back(id, std::move(buffers));
902 
903  DoConnect();
904  }
905  else
906  {
907  BOND_LOG(LOG_DEBUG, "EpoxyConnection",
908  "Reconnect for packet (id: " << id << ") to " << m_address << " is disabled");
909 
910  BOOST_VERIFY(m_reading.exchange(false));
911  BOOST_VERIFY(m_sending.exchange(false));
912 
913  //
914  // Reconnect is disabled, notify upper layer that this packed is dropped
915  //
916  m_handler->PacketSent(false, id);
917 
918  //
919  // Drop all enqued packets so far
920  DropQueue();
921  }
922  }
