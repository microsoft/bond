166 class service::unary_call_data : io_manager_tag
167 {
168 public:
169 template <typename Request, typename Response>
170 unary_call_data(
171 service& service,
172 int methodIndex,
173 const std::function<void(unary_call<Request, Response>)>& cb)
174 : _service{ service },
175 _methodIndex{ methodIndex },
176 _invoke{ std::bind(&unary_call_data::invoke<Request, Response>, this, cb) },
177 _receivedCall{}
178 {
179 BOOST_ASSERT(cb);
180 queue_receive();
181 }
182
183 private:
184 template <typename Request, typename Response>
185 void invoke(const std::function<void(unary_call<Request, Response>)>& callback)
186 {
187 // TODO: Use lambda with move-capture when allowed to use C++14.
188 _service.scheduler()(std::bind(
189 [](const decltype(callback)& cb, boost::intrusive_ptr<unary_call_impl>& receivedCall)
190 {
191 cb(unary_call<Request, Response>{ std::move(receivedCall) });
192 },
193 callback,
194 queue_receive()));
195 }
196
197 void invoke(bool ok) override
198 {
199 if (ok)
200 {
201 BOOST_ASSERT(_invoke);
202 _invoke();
203 }
204 }

206 boost::intrusive_ptr<unary_call_impl> queue_receive()
207 {
208 boost::intrusive_ptr<unary_call_impl> receivedCall{ _receivedCall.release() };
209
210 // create new state for the next request that will be received
211 _receivedCall.reset(new unary_call_impl{});
212
213 _service.queue_receive(
214 _methodIndex,
215 &_receivedCall->context(),
216 &_receivedCall->request_buffer(),
217 &_receivedCall->responder(),
218 tag());
219
220 return receivedCall;
221 }

231 std::unique_ptr<unary_call_impl> _receivedCall;
